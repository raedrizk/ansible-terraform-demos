#GCP
curl -O https://dl.google.com/dl/cloudsdk/channels/rapid/downloads/google-cloud-cli-383.0.1-linux-x86_64.tar.gz
tar -xzf google-cloud-cli-383.0.1-linux-x86_64.tar.gz 
./google-cloud-sdk/install.sh 
./google-cloud-sdk/bin/gcloud auth activate-service-account --key-file=<PATH_TO_SERVICE_ACCOUNT_JSON>
./google-cloud-sdk/bin/gcloud compute images list | grep rhel


mkdir /root/.gcp
cp <PATH_TO_SERVICE_ACCOUNT_JSON> /root/.gcp
echo "export GOOGLE_APPLICATION_CREDENTIALS=\"/root/.gcp/serviceaccount.json\"" > /etc/profile.d/gcp.sh
